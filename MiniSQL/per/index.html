    Planner and Executor

    Planner & Executor 设计报告

    一、模块描述

    本实验主要包括Planner和Executor两部分。Planner的主要功能是将解释器（Parser）生成的语法树，改写成数据库可以理解的数据结构。在这个过程中，我们会将所有sql语句中的标识符（Identifier）解析成没有歧义的实体，即各种C++的类，并通过Catalog Manager 提供的信息生成执行计划。Executor遍历查询计划树，将树上的 PlanNode 替换成对应的 Executor，随后调用 Record Manager、Index Manager 和 Catalog Manager 提供的相应接口进行执行，并将执行结果返回给上层模块。

    二、主要功能

    对于简单的语句例如show databases，drop table 等，生成的语法树也非常简单。以show databases为例，对应的语法树只有单节点kNodeShowDB，表示展示所有数据库。此时无需传入Planner生成执行计划，我们直接调用对应的执行函数执行即可。

    对于复杂的语句，生成的语法树需传入Planner生成执行计划，并交由Executor进行执行。Planner需要先遍历语法树，调用Catalog Manager 检查语法树中的信息是否正确，如表、列是否存在，谓词的值类型是否与column类型对应等等，随后将这些词语抽象成相应的表达式。解析完成后，Planner根据改写语法树后生成的可以理解的Statement结构，生成对应的Plannode，并将Planndoe交由executor进行执行。

    EXcutor采用的是Iterator Model：执行引擎会将整个 SQL 构建成一个 Operator 树，查询树自顶向下的调用接口，数据则自底向上的被拉取处理。每一种操作会抽象为一个 Operator，每个算子都有 Init() 和 Next() 两个方法。Init() 对算子进行初始化工作。Next() 则是向下层算子请求下一条数据。当 Next() 返回 false 时，则代表下层算子已经没有剩余数据，迭代结束。

    三、设计思路

    1.SeqScanExecutor

    SeqScanExecutor 对表执行一次顺序扫描，一次Next()方法返回一个符合谓词条件的行。顺序扫描的表名和谓词（Predicate）由SeqScanPlanNode 指定。

    需要注意：

    • 最后根据OutputSchema来获得过滤后的column，即需要输出的column。

bool SeqScanExecutor::Next(Row *row, RowId *rid) {
    do{
        if(table_iter_ == table_info_->GetTableHeap()->End())
            return false;
        *row = *table_iter_;
        *rid = row->GetRowId();
        table_iter_++;
    }while(plan_->GetPredicate() != nullptr && !plan_->GetPredicate()->Evaluate(row).CompareEquals(Field(kTypeInt, 1)));

    auto schema = plan_->OutputSchema();
    std::vector<Field> values;
    for(auto &col : schema->GetColumns()){
        uint32_t idx;
        table_info_->GetSchema()->GetColumnIndex(col->GetName(),idx);
        values.emplace_back(*row->GetField(idx));
    }
    *row = Row{values};
    return true;
}
    2.IndexScanExecutor

    IndexScanExecutor 对表执行一次带索引的扫描，一次Next()方法返回一个符合谓词条件的行。为简单起见，IndexScan仅支持单列索引。当Planner检测到select的谓词中的列上存在索引，而且索引只包含该列时，会生成IndexScanPlan，其他情况则生成SeqScanPlan。同时，为简单起见，IndexScan不支持谓词中存在or的情况（例如a=1 or b=2，此时a列存在索引，b列不存在索引，我们仍需顺序扫描表来找出符合b=2条件的列）。因此，在IndexScanExecutor中你只需要考虑单列索引，并且不用考虑谓词中存在or的情况。

    需要注意：

    • 调用index中的ScanKey可以返回单列索引需要的结果
    • 所有索引的结果取交集就是最终结果
    • 如果某个谓词没有索引还需要调用GetPredicate来判断
    • 所有索引的结果取交集的查询效率其实是比较低下的

#include <algorithm>

// Student added function
pair<string, Field> FindIndexVal(AbstractExpression *node, uint32_t col_idx) {
  ComparisonExpression *comparisonNode = nullptr;
  string comparisonType;
  ColumnValueExpression *col = nullptr;
  ConstantValueExpression *cons = nullptr;
  auto type = node->GetType();
  if (type == ExpressionType::LogicExpression) {
    for (const auto &child : node->GetChildren()) {
      pair<string, Field> field = FindIndexVal(&*(child), col_idx);
      if (!field.second.IsNull()) {
        return field;
      }
    }
  } else if (type == ExpressionType::ComparisonExpression) {
    comparisonNode = dynamic_cast<ComparisonExpression *>(&*node);
    comparisonType = comparisonNode->GetComparisonType();
    if (node->GetChildAt(0)->GetType() == ExpressionType::ColumnExpression && node->GetChildAt(0)) {
      col = dynamic_cast<ColumnValueExpression *>(&*node->GetChildAt(0));
      if (col->GetColIdx() == col_idx) {
        cons = dynamic_cast<ConstantValueExpression *>(&*node->GetChildAt(1));
        return pair<string, Field>(comparisonType, Field(cons->val_));
      } else {
        return pair<string, Field>("", Field(TypeId::kTypeInvalid));
      }
    }
  } else
    return pair<string, Field>("", Field(TypeId::kTypeInvalid));
}

bool RowIdComp(RowId a, RowId b) {
  return a.GetPageId() > b.GetPageId() || (a.GetPageId() == b.GetPageId() && a.GetSlotNum() > b.GetSlotNum());
}

void IndexScanExecutor::Init() {
  pair<string, Field> *res;
  unordered_map<IndexInfo *, pair<string, Field *>> index_val;
  for (auto index : plan_->indexes_) {
    auto col_idx = index->GetIndexKeySchema()->GetColumn(0)->GetTableInd();
    AbstractExpression *node = &*(plan_->filter_predicate_);
    res = new pair<string, Field>(FindIndexVal(node, col_idx));
    index_val[index].first = res->first;
    index_val[index].second = &res->second;
  }

  for (auto index : plan_->indexes_) {
    vector<Field> row;
    vector<RowId> results, prevRes;
    index_results_.swap(prevRes);
    row.emplace_back(*index_val[index].second);
    index->GetIndex()->ScanKey(Row(row), results, nullptr, index_val[index].first);
    sort(results.begin(), results.end(), RowIdComp);
    if (*plan_->indexes_.begin() != index) {
      std::set_intersection(results.begin(), results.end(), prevRes.begin(), prevRes.end(),
                            std::back_inserter(index_results_), RowIdComp);
    } else {
      index_results_ = results;
    }
  }
  iter_ = index_results_.begin();
}

bool IndexScanExecutor::Next(Row *row, RowId *rid) {
  while (iter_ != index_results_.end()) {
    Row *tuple = new Row(*iter_);
    table_info_->GetTableHeap()->GetTuple(tuple, nullptr);
    if (plan_->GetPredicate()->Evaluate(tuple).CompareEquals(Field(kTypeInt, 1))) {
      vector<Field> fields;
      for (auto column : original_schema_->GetColumns()) {
        for (auto target : plan_->OutputSchema()->GetColumns()) {
          if (!target->GetName().compare(column->GetName())) {
            fields.push_back(*tuple->GetField(column->GetTableInd()));
          }
        }
      }
      *row = Row(fields);
      row->SetRowId(*iter_);
      delete tuple;
      *rid = *iter_;
      ++iter_;
      return true;
    }
    delete tuple;
    iter_++;
  }
  return false;
}
    3.InsertExecutor

    InsertExecutor 将行插入表中并更新索引。目前，Parser只支持直接插入insert into t1 values(1, "aaa", null, 2.33);，不支持 INSERT INTO .. SELECT ...的语法。要插入的值通过ValueExecutor生成对应的行，随后被拉取到InsertExecutor中。

    需要注意：

    • 插入需要检查primary key和unique是否重复，这在一定程度上降低了插入效率
    • 插入的同时要对索引进行更新，也会影响到插入效率

    : AbstractExecutor(exec_ctx), plan_(plan), child_executor_(std::move(child_executor)) {}

  child_executor_->Init();

bool InsertExecutor::Next([[maybe_unused]] Row *row, RowId *rid) {
  string table_name_=plan_->GetTableName();
  auto catalog_mgr_=exec_ctx_->GetCatalog();
  TableInfo* table_info_;
  catalog_mgr_->GetTable(table_name_, table_info_);
  auto table_heap_=table_info_->GetTableHeap();
  std::vector<IndexInfo* >table_indexes_;
  catalog_mgr_->GetTableIndexes(table_name_, table_indexes_);
    if(child_executor_->Next(row, rid)){
      for(auto index: table_indexes_){
        if(index->GetIndexName().find("UNIQUE_")==0&&index->GetIndexKeySchema()->GetColumnCount()==1){  //index字段包含unique就行
          uint32_t col_ind=index->GetIndexKeySchema()->GetColumn(0)->GetTableInd(); //找到索引对应的列
          Field* val=row->GetField(col_ind);
          std::vector<Field> key_tmp{Field(*val)};
          Row key=Row(key_tmp);
          std::vector<RowId>res;
          index->GetIndex()->ScanKey(key, res, exec_ctx_->GetTransaction(), "="); //找之前的有没有相同的
          if(!res.empty()){ //之前有相同的
            std::cout<<"uni constraint conflict"<<std::endl;
            return false;
          }
        }
        if(index->GetIndexName().compare("PRIMARY_")==0&&index->GetIndexKeySchema()->GetColumnCount()==1){
          uint32_t col_ind=index->GetIndexKeySchema()->GetColumn(0)->GetTableInd(); //找到索引对应的列
          Field* val=row->GetField(col_ind);
          std::vector<Field> key_tmp{Field(*val)};
          Row key=Row(key_tmp);
          std::vector<RowId>res;
          index->GetIndex()->ScanKey(key, res, exec_ctx_->GetTransaction(), "="); //找之前的有没有相同的
          if(!res.empty()){ //之前有相同的
            std::cout<<"primary key constraint conflict"<<std::endl;
            return false;
          }
        }
      }
      bool is_inserted=table_heap_->InsertTuple(*row, exec_ctx_->GetTransaction());
      if(!is_inserted)
        return false;
      for(auto ind: table_indexes_){
        uint32_t col_index=ind->GetIndexKeySchema()->GetColumn(0)->GetTableInd();
        Field* val=row->GetField(col_index);
        std::vector<Field> key_tmp{Field(*val)};
        Row key=Row(key_tmp);
        ind->GetIndex()->InsertEntry(key, row->GetRowId(), exec_ctx_->GetTransaction());
      }
      *rid=row->GetRowId();
      //std::cout<<"success to insert to"<<table_name_<<std::endl;
      return true;
    }
  return false;
}
    4.UpdataExecutor

    UpdateExecutor 修改指定表中的现有行并更新其索引。UpdatePlanNode 将有一个 SeqScanPlanNode 作为其子节点，要更新的值通过SeqScanExecutor提供。

    需要注意：

    • 插入实际上就是把原来的row替换成新的row
    • 对索引的维护也是将原先的key删除，再插入新的key

void UpdateExecutor::Init() {
  child_executor_->Init();
  catalog_mgr_=exec_ctx_->GetCatalog();
  table_name_=plan_->GetTableName();
  catalog_mgr_->GetTable(table_name_, table_info_);
  table_heap_=table_info_->GetTableHeap();
  catalog_mgr_->GetTableIndexes(table_name_, table_indexes_);
  schema_=table_info_->GetSchema();
  update_attrs_=plan_->GetUpdateAttr();
  column_cnt=schema_->GetColumnCount();
}

bool UpdateExecutor::Next([[maybe_unused]] Row *row, RowId *rid) {
  if(child_executor_->Next(row, rid)){
    Row update_row=GenerateUpdatedTuple(*row);
    bool is_updated=table_heap_->UpdateTuple(update_row, *rid, exec_ctx_->GetTransaction());
    if(!is_updated)
      return false;
    for(auto index: table_indexes_){
      Row key_row;
      row->GetKeyFromRow(schema_, index->GetIndexKeySchema(), key_row);
      index->GetIndex()->RemoveEntry(key_row, *rid, exec_ctx_->GetTransaction());
      update_row.GetKeyFromRow(schema_, index->GetIndexKeySchema(), key_row);
      index->GetIndex()->InsertEntry(key_row, *rid, exec_ctx_->GetTransaction());
    }
    return true;
  }
  return false;
}

Row UpdateExecutor::GenerateUpdatedTuple(const Row &row) {
  //auto update_attrs=plan_->GetUpdateAttr();
  std::vector<Field> update_row;
  for(uint32_t i=0; i<column_cnt; i++){
    if(update_attrs_.find(i)==update_attrs_.end()){ //不是要更新的
      update_row.emplace_back(*row.GetField(i));
    }
    else{ //是要更新的field就新建一个
      Field new_field=update_attrs_.at(i)->Evaluate(nullptr);
      update_row.emplace_back(new_field);
    }
  }
  return Row(update_row);
  //return Row();
}

  catalog_mgr_->GetTable(table_name_, table_info_);
  table_heap_=table_info_->GetTableHeap();
  catalog_mgr_->GetTableIndexes(table_name_, table_indexes_);

bool DeleteExecutor::Next([[maybe_unused]] Row *row, RowId *rid) {
  if(child_executor_->Next(row, rid)){
    bool is_deleted=table_heap_->MarkDelete(*rid, exec_ctx_->GetTransaction());
    if(!is_deleted)
      return false;
    for(auto ind: table_indexes_){
      Row key_row;
      row->GetKeyFromRow(table_info_->GetSchema(), ind->GetIndexKeySchema(), key_row);
      ind->GetIndex()->RemoveEntry(key_row, *rid, exec_ctx_->GetTransaction());
    }
    return true;
  }
  return false;
}

    7.DropDatabase

    dberr_t ExecuteEngine::ExecuteDropDatabase(pSyntaxNode ast, ExecuteContext *context) {
#ifdef ENABLE_EXECUTE_DEBUG
  LOG(INFO) << "ExecuteDropDatabase" << std::endl;
#endif
    if(ast->child_ == nullptr)
        return DB_FAILED;
    std::string db_name(ast->child_->val_);
    if(dbs_[db_name] == nullptr)
    {
        std::cout << "database " << db_name << " exists." << endl;
        return DB_FAILED;
    }
    if(current_db_ == db_name)
        current_db_ = "";
    auto target_db = dbs_[db_name];
    target_db->~DBStorageEngine();
    dbs_.erase(db_name);
    fstream out;
    out.open("/mnt/d/dbs_name.txt"); // 可修改路径
    out << dbs_.size() << std::endl;
    for(auto db : dbs_){
        out << db.first << std::endl;
    }
    std::cout << "Drop success" << std::endl;
    return DB_SUCCESS;
}

    11.CreateTable

    翻译语法树，利用目录页的CreateTable函数来创建表

    需要注意：

    • 要拿到语法树中的primary key和unique信息
    • 给primary key和unique都创建索引
    dberr_t ExecuteEngine::ExecuteCreateTable(pSyntaxNode ast, ExecuteContext *context) {
#ifdef ENABLE_EXECUTE_DEBUG
  LOG(INFO) << "ExecuteCreateTable" << std::endl;
#endif
    if(dbs_[current_db_] == nullptr){
        std::cout << "no database selected" << std::endl;
        return DB_FAILED;
    }
    pSyntaxNode ptr = ast->child_;
    string table_name(ptr->val_);
    ptr = ptr->next_->child_;
    std::vector<Column*> columns;
    std::vector<std::string> uni;
    std::vector<std::string> pri;
    string column_name;
    string column_type;
    TypeId type;
    uint32_t  column_length = 4;
    for(uint32_t i = 0;ptr != nullptr;i++,ptr = ptr->next_){
        if(ptr->val_ == nullptr){ //非 unique
            column_name = ptr->child_->val_;
            column_type = ptr->child_->next_->val_;
            if(column_type == "int"){
                type = kTypeInt;
            }
            else if(column_type == "float"){
                type = kTypeFloat;
            }
            else if(column_type == "char"){
                type = kTypeChar;
                column_length = atoi(ptr->child_->next_->child_->val_);
                if(column_length <= 0 || strchr(ptr->child_->next_->child_->val_,'.') != nullptr){
                    std::cout << "char invalid" << std::endl;
                    return DB_FAILED;
                }
            }
            else{
                type = kTypeInvalid;
                cout << "Invalid type" << std::endl;
                return DB_FAILED;
            }
            Column* column_ptr;
            if(type == kTypeInt || type == kTypeFloat)
                column_ptr = new Column(column_name,type,i, false, false);
            else
                column_ptr = new Column(column_name, type, column_length, i, false, false);
            columns.push_back(column_ptr);
        }
        else if(!strcmp(ptr->val_,"unique")) {
            column_name = ptr->child_->val_;
            column_type = ptr->child_->next_->val_;
            if(column_type == "int"){
                type = kTypeInt;
            }
            else if(column_type == "float"){
                type = kTypeFloat;
            }
            else if(column_type == "char"){
                type = kTypeChar;
                column_length = atoi(ptr->child_->next_->child_->val_);
                if(column_length <= 0 || strchr(ptr->child_->next_->child_->val_,'.') != nullptr){
                    std::cout << "char invalid" << std::endl;
                    return DB_FAILED;
                }
            }
            else{
                type = kTypeInvalid;
                cout << "Invalid type" << std::endl;
                return DB_FAILED;
            }
            uni.emplace_back(column_name);
            Column* column_ptr;
            if(type == kTypeInt || type == kTypeFloat)
                column_ptr = new Column(column_name,type,i, false, true);
            else
                column_ptr = new Column(column_name, type, column_length, i, false, true);
            columns.push_back(column_ptr);
        }
        else if(!strcmp(ptr->val_,"primary keys")){
            for(auto pri_name = ptr->child_;pri_name != nullptr;pri_name = pri_name->next_){
                pri.emplace_back(pri_name->val_);
            }
            for(auto pri_name : pri){
                for(auto column : columns){
                    if(column->GetName() == pri_name){
                        if(pri.size() == 1){
                            column->SetUnique(true);
                        }
                        column->SetNullable(false);
                        break;
                    }
                }
            }
        }
    }
    auto* schema = new Schema(columns);
    TableInfo* table_info = nullptr;
    auto mgr = dbs_[current_db_]->catalog_mgr_;
    dberr_t create = mgr->CreateTable(table_name,schema, nullptr,table_info);
    if(create != DB_SUCCESS)
        return create;
    for(int i = 0;i < uni.size();i++){  //unique需要创建索引
        std::vector<string> index_key;
        index_key.emplace_back(uni.at(i));
        auto index_name = "UNIQUE_" + to_string(i);
        auto index_info = IndexInfo::Create();
        auto res = dbs_[current_db_]->catalog_mgr_->CreateIndex
                (table_name,index_name,index_key, nullptr,index_info,"bptree");
        if(res != DB_SUCCESS)
            return res;
    }
    if(pri.size() == 1){
        string index_name = "PRIMARY_";
        auto index_info = IndexInfo::Create();
        auto res = dbs_[current_db_]->catalog_mgr_->CreateIndex
                (table_name,index_name,pri, nullptr,index_info,"bptree");
        if(res != DB_SUCCESS)
            return res;
    }
    cout << "create table success" << endl;
    return DB_SUCCESS;
}

    15.DropIndex

    删除索引，嵌套遍历每一个table中的index，找到特定的index后删除。

    dberr_t ExecuteEngine::ExecuteDropIndex(pSyntaxNode ast, ExecuteContext *context) {
#ifdef ENABLE_EXECUTE_DEBUG
  LOG(INFO) << "ExecuteDropIndex" << std::endl;
#endif
    auto idx_node = ast->child_;
    string idx_name = idx_node->val_;
    auto cata_manager = dbs_[current_db_]->catalog_mgr_;
    std::vector<TableInfo*> table_infos;
    auto res = cata_manager->GetTables(table_infos);
    if(res != DB_SUCCESS){
        return res;
    }
    for(auto table_info : table_infos){
        string table_name = table_info->GetTableName();
        std::vector<IndexInfo*> index_infos;
        res = cata_manager->GetTableIndexes(table_name,index_infos);
        if(res != DB_SUCCESS){
            return res;
        }
        for(auto index_info : index_infos){
            if(index_info->GetIndexName() == idx_name){
                res = cata_manager->DropIndex(table_info->GetTableName(),idx_name);
                if(res != DB_SUCCESS){
                    return res;
                }
                cout << "drop index success"<< endl;
                return  DB_SUCCESS;
            }
        }
    }
    return DB_FAILED;
}
    16.ExecFile

    读入sql文本文件，执行文本中给出的操作。直接对照main函数来写。

    dberr_t ExecuteEngine::ExecuteExecfile(pSyntaxNode ast, ExecuteContext *context) {
#ifdef ENABLE_EXECUTE_DEBUG
  LOG(INFO) << "ExecuteExecfile" << std::endl;
#endif
    char* file_name = ast->child_->val_;
    FILE* file = fopen(file_name, "r");
    if (file == nullptr) {
        cout << "File " << file_name << " not found!" << endl;
        return DB_FAILED;
    }
    char input[1024];
    while(!feof(file)) {
        memset(input, 0, 1024);
        int i = 0;
        char ch;
        while (!feof(file) && (ch = getc(file)) != ';') {
            input[i++] = ch;
        }
        if (feof(file)) continue;
        input[i] = ch;
        YY_BUFFER_STATE bp = yy_scan_string(input);
        if (bp == nullptr) {
            LOG(ERROR) << "Failed to create yy buffer state." << std::endl;
            exit(1);
        }
        yy_switch_to_buffer(bp);

        // init parser module
        MinisqlParserInit();

        // parse
        yyparse();

        // parse result handle
        if (MinisqlParserGetError()) {
            // error
            printf("%s\n", MinisqlParserGetErrorMessage());
        }

        auto result = this->Execute(MinisqlGetParserRootNode());

        // clean memory after parse
        MinisqlParserFinish();
        yy_delete_buffer(bp);
        yylex_destroy();

        ExecuteInformation(result);
    }
    return DB_SUCCESS;
}
