    Catalog Manager

    Catalog Manager 设计报告

    一、模块描述

    Catalog Manager 负责管理和维护数据库的所有模式信息，包括：

    • 数据库中所有表的定义信息，包括表的名称、表中字段（列）数、主键、定义在该表上的索引。
    • 表中每个字段的定义信息，包括字段类型、是否唯一等。
    • 数据库中所有索引的定义，包括所属表、索引建立在那个字段上等。

    这些模式信息在被创建、修改和删除后还应被持久化到数据库文件中。此外，Catalog Manager还需要为上层的执行器Executor提供公共接口以供执行器获取目录信息并生成执行计划。

    二、主要功能

    CatalogManager类应具备维护和持久化数据库中所有表和索引的信息的功能。CatalogManager能够在数据库实例（DBStorageEngine）初次创建时（init = true）初始化元数据；并在后续重新打开数据库实例时，从数据库文件中加载所有的表和索引信息，构建TableInfo和IndexInfo信息置于内存中。此外，CatalogManager类还需要对上层模块提供对指定数据表的操作方式，如CreateTable、GetTable、GetTables、DropTable、GetTableIndexes；对上层模块提供对指定索引的操作方式，如CreateIndex、GetIndex、DropIndex。

    三、设计思路

    1.序列化

    需要了解Catalog, Table, Index的元信息的序列化过程，我们需要实现的是求序列化的大小。只用通过阅读元信息的SerializeTo和DeserializeFrom函数就能知道元信息由哪些部分组成，并求得大小。这并不是本章的难点，不过多赘述。

    2.Indexes对象初始化函数Init

    分成三步实现:

    • 初始化Index的元信息和table_info
    • 做key到key_schema的映射
    • 调用CreateIndex函数来创建索引

    比较简单，不赘述。

    void Init(IndexMetadata *meta_data, TableInfo *table_info, BufferPoolManager *buffer_pool_manager) {
  // Step1: init index metadata and table info
  // Step2: mapping index key to key schema
  // Step3: call CreateIndex to create the index
  meta_data_=meta_data;
  key_schema_=Schema::ShallowCopySchema(table_info->GetSchema(), meta_data->GetKeyMapping());
  index_=CreateIndex(buffer_pool_manager, "bptree");
}
    3.CatalogManager的构造函数

    在上层对数据库进行操作时，会生成一个CatalogManager的实例，对底层数据进行访问并读入catalog。

    需要注意如下几点：

    • init表明是否已经存在Meta的实例，如果init为ture，需要为CatalogMeta创建一个新实例。
    • 分别初始化table_info,table_meta,index_info,index_meta

    代码如下:

    CatalogManager::CatalogManager(BufferPoolManager *buffer_pool_manager, LockManager *lock_manager,
                               LogManager *log_manager, bool init)
        : buffer_pool_manager_(buffer_pool_manager), lock_manager_(lock_manager), log_manager_(log_manager) {
    std::atomic_init(&next_table_id_,0);
    std::atomic_init(&next_index_id_,0);
    if (init) {
        catalog_meta_ = CatalogMeta::NewInstance();
    } else {
        auto meta_page = buffer_pool_manager_->FetchPage(CATALOG_META_PAGE_ID);
        catalog_meta_ = CatalogMeta::DeserializeFrom(meta_page->GetData());
        // 加载table
        for (auto it : catalog_meta_->table_meta_pages_) {
            auto table_meta_page = buffer_pool_manager_->FetchPage(it.second);
            TableMetadata *table_meta;
            TableMetadata::DeserializeFrom(table_meta_page->GetData(), table_meta);
            auto table_heap = TableHeap::Create(buffer_pool_manager_, table_meta->GetFirstPageId(), table_meta->GetSchema(),
                                                log_manager_, lock_manager_);
            table_names_[table_meta->GetTableName()] = table_meta->GetTableId();
            TableInfo *table_info = TableInfo::Create();
            table_info->Init(table_meta, table_heap);
            tables_[table_meta->GetTableId()] = table_info;
        }
        // 加载index
        for (auto it : catalog_meta_->index_meta_pages_) {
            auto index_meta_page = buffer_pool_manager_->FetchPage(it.second);
            IndexMetadata *index_meta;
            IndexMetadata::DeserializeFrom(index_meta_page->GetData(), index_meta);
            index_names_[tables_[index_meta->GetTableId()]->GetTableName()][index_meta->GetIndexName()] = index_meta->GetIndexId();
            auto index_info = IndexInfo::Create();
            index_info->Init(index_meta, tables_[index_meta->GetTableId()], buffer_pool_manager_);
            indexes_[index_meta->GetIndexId()] = index_info;
        }
    }
}
    4.CreateTable

    创建一个表，需要完成以下操作：

    • table_info的初始化（包括table_meta和table_heap）
    • 在catalog中为该表分配一个新页，并更新catalog中的tables_和table_names
    • 最后将catalog的meta再flush一遍
    dberr_t CatalogManager::CreateTable(const string &table_name, TableSchema *schema,
                                    Transaction *txn, TableInfo *&table_info) {
    if(table_names_.find(table_name) != table_names_.end())
        return DB_ALREADY_EXIST;
    table_names_[table_name] = next_table_id_;  // 更新table_names_

    page_id_t page_id;
    auto page = buffer_pool_manager_->NewPage(page_id); // 分配新的page
    catalog_meta_->table_meta_pages_[next_table_id_] = page_id; // 更新catalog_meta_

    table_info = TableInfo::Create();
    auto new_schema = Schema::DeepCopySchema(schema);
    auto table_heap = TableHeap::Create(buffer_pool_manager_, new_schema, txn, log_manager_, lock_manager_);
    auto table_meta = TableMetadata::Create(next_table_id_, table_name, table_heap->GetFirstPageId(), new_schema);
    table_info->Init(table_meta, table_heap);
    tables_[next_table_id_] = table_info; // 更新tables_

    next_table_id_ = catalog_meta_->GetNextTableId(); // 更新next_table_id_
    table_meta->SerializeTo(page->GetData());
    buffer_pool_manager_->UnpinPage(page_id, true);
    FlushCatalogMetaPage();
    return DB_SUCCESS;
}

    dberr_t CatalogManager::GetTables(vector<TableInfo *> &tables) const {
    if(tables_.empty()) return DB_FAILED;
    // 遍历tables_，将所有的table_info放入tables

    7.CreateIndex

    创建一个索引，需要完成以下操作：

    • index_info的初始化（包括table_meta和index_）
    • 在catalog中为该表分配一个新页，并更新catalog中的indexes_和index_names
    • 注意需要物理上插入index，调用index的InsertEntry来实现
    • 最后将catalog的meta再flush一遍
    dberr_t CatalogManager::CreateIndex(const std::string &table_name, const string &index_name,
                                    const std::vector<std::string> &index_keys, Transaction *txn,
                                    IndexInfo *&index_info, const string &index_type) {
    if(table_names_.find(table_name) == table_names_.end())
        return DB_TABLE_NOT_EXIST;
    auto& index_map = index_names_[table_name];
    if(index_map.find(index_name) != index_map.end())
        return DB_INDEX_ALREADY_EXIST;
    // 为新的index分配page
    auto table_id = table_names_[table_name];
    std::vector<uint32_t> key_map;
    auto table_info = tables_[table_id];
    auto schema = table_info->GetSchema();
    for(auto &it: index_keys){
        uint32_t index;
        if(schema->GetColumnIndex(it, index) == DB_COLUMN_NAME_NOT_EXIST) return DB_COLUMN_NAME_NOT_EXIST;
        key_map.push_back(index);
    } // 将index_keys转换为key_map
    auto index_meta = IndexMetadata::Create(next_index_id_, index_name, table_id, key_map);
    // 创建index_meta
    page_id_t page_id;
    auto page = buffer_pool_manager_->NewPage(page_id);
    if(page == nullptr)
        return DB_FAILED;
    catalog_meta_->index_meta_pages_[next_index_id_] = page_id;
    index_map[index_name] = next_index_id_;
    index_info = IndexInfo::Create();
    index_info->Init(index_meta, table_info, buffer_pool_manager_);
    indexes_[next_index_id_] = index_info; // 更新indexes_
    // 更新catalog_meta_
    next_index_id_ = catalog_meta_->GetNextIndexId();
    index_meta->SerializeTo(page->GetData());
    buffer_pool_manager_->UnpinPage(page_id, true);

    //需要物理插入index
    auto table_heap = table_info->GetTableHeap();
    vector<Field> values;
    for (auto it = table_heap->Begin(nullptr); it != table_heap->End(); it++) {
        values.clear();
        for (auto pos : key_map) {
            values.push_back(*(it->GetField(pos)));
        }
        Row row{values};
        index_info->GetIndex()->InsertEntry(row, it->GetRowId(), nullptr);
    }
    FlushCatalogMetaPage();

    return DB_SUCCESS;
}
    8.GetIndex

    通过table_name,index_name得到index_info

    dberr_t CatalogManager::GetIndex(const std::string &table_name, const std::string &index_name,
                                 IndexInfo *&index_info) const {
    if(table_names_.find(table_name)==table_names_.end())
        return DB_TABLE_NOT_EXIST;
    if(index_names_.at(table_name).find(index_name)==index_names_.at(table_name).end())
        return DB_INDEX_NOT_FOUND;
    auto index_id=index_names_.at(table_name).at(index_name);
    index_info=indexes_.at(index_id);
    return DB_SUCCESS;
}

    10.DropTable

    删除table，需要删除：

    • table_names_
    • tables_
    • Table_meta_pages
    dberr_t CatalogManager::DropTable(const string &table_name) {
    if(table_names_.find(table_name)==table_names_.end())
        return DB_TABLE_NOT_EXIST;
    auto table_id=table_names_[table_name];
    table_names_.erase(table_name);

    auto page_id=catalog_meta_->table_meta_pages_[table_id];
    catalog_meta_->table_meta_pages_.erase(table_id);

    buffer_pool_manager_->DeletePage(page_id);  //删除page
    tables_.erase(table_id);
    FlushCatalogMetaPage();
    return DB_SUCCESS;
}
    11.DropIndex

    删除index，需要删除：

    • index_names
    • indexes_
    • index_meta_pages_
    dberr_t CatalogManager::DropIndex(const string &table_name, const string &index_name) {
    if(table_names_.find(table_name) == table_names_.end())
        return DB_TABLE_NOT_EXIST;
    if(index_names_.at(table_name).find(index_name) == index_names_.at(table_name).end())
        return DB_INDEX_NOT_FOUND;

    auto index_id = index_names_.at(table_name).at(index_name);
    index_names_.at(table_name).erase(index_name); // 删除index_name
    auto page_id = catalog_meta_->index_meta_pages_[index_id];
    catalog_meta_->index_meta_pages_.erase(index_id);

    buffer_pool_manager_->DeletePage(page_id); // 删除index_meta_page
    indexes_.erase(index_id); // 删除index
    FlushCatalogMetaPage();
    return DB_SUCCESS;
}
    12.FlushCatalogMetaPage

    每次需要更新catalog都需要flush

    dberr_t CatalogManager::FlushCatalogMetaPage() const {
    auto page=buffer_pool_manager_->FetchPage(CATALOG_META_PAGE_ID);
    catalog_meta_->SerializeTo(page->GetData());
    buffer_pool_manager_->UnpinPage(CATALOG_META_PAGE_ID, true);
    if(!buffer_pool_manager_->FlushPage(CATALOG_META_PAGE_ID))
        return DB_FAILED;
    return DB_SUCCESS;
}
    13.LoadTable(unused)

    根据page id从meta data中拿到table的信息并在catalog中更新

    dberr_t CatalogManager::LoadTable(const table_id_t table_id, const page_id_t page_id) {
    if(tables_.find(table_id)==tables_.end())
        return DB_TABLE_NOT_EXIST;

    auto table_meta_page=buffer_pool_manager_->FetchPage(page_id);
    TableMetadata *table_meta;
    TableMetadata::DeserializeFrom(table_meta_page->GetData(), table_meta);
    table_names_[table_meta->GetTableName()]=table_id;//更新table_names_
    TableHeap* table_heap=TableHeap::Create(buffer_pool_manager_, page_id, table_meta->GetSchema(), log_manager_, lock_manager_);
    TableInfo* table_info=TableInfo::Create();
    table_info->Init(table_meta, table_heap);
    tables_[table_id]=table_info; //更新tables_
    catalog_meta_->table_meta_pages_[table_id]=page_id; //更新catalog_meta_
    FlushCatalogMetaPage();
    return DB_SUCCESS;
}
    14.LoadIndex(unused)

    根据index id从meta data中拿到index的信息并在catalog中更新

    dberr_t CatalogManager::LoadIndex(const index_id_t index_id, const page_id_t page_id) {
    if(indexes_.find(index_id) != indexes_.end()){
        return DB_INDEX_NOT_FOUND;
    }

    auto index_meta_page = buffer_pool_manager_->FetchPage(page_id);
    IndexMetadata *index_meta;
    IndexMetadata::DeserializeFrom(index_meta_page->GetData(),index_meta);
    auto table_name = tables_[index_meta->GetTableId()]->GetTableName();
    index_names_[table_name][index_meta->GetIndexName()] = index_id; //载入index_names_

    auto index_info = IndexInfo::Create();
    index_info->Init(index_meta,tables_[index_meta->GetTableId()],buffer_pool_manager_);
    indexes_[index_id] = index_info; //载入indexes_

    catalog_meta_->index_meta_pages_[index_id] = page_id; //载入index_meta_pages_
    FlushCatalogMetaPage();
    return DB_SUCCESS;
}
    15.GetTable

    根据table_id在目录中拿到table_info的指针

    dberr_t CatalogManager::GetTable(const table_id_t table_id, TableInfo *&table_info) {
    if(catalog_meta_->table_meta_pages_.find(table_id) == catalog_meta_->table_meta_pages_.end()){
        return DB_TABLE_NOT_EXIST;
    }
    table_info = tables_[table_id];
    return DB_SUCCESS;
}
