    类似于DAI的stable coin构建

    记录了我构建类似于DAI的stable coin的全流程。

    1 稳定币概述

    项目中的稳定币是参考了DAI，1:1与美元挂钩的稳定币。用了liquidate的算法来保证1:1的稳定性。其中liquidate算法也是该项目中的重点，即当用户的质押物不足以维持其取出的stable coin的数量时(低于health factor)，则其他用户可以对该用户进行清算，即通过较便宜的价格购买低于health factor用户的质押物，以此来维持整个DeFi始终保持over-collateralized(质押物的价值 > 已经铸造的稳定币的价值)。通过维持整个DeFi的over-collateralized的性质，来维持1:1的稳定性。liquidate函数会在后续展开介绍。

    2 构建ERC20

    代码：DSC

    用了openzepplin库来快速构建ERC20，稳定币主要功能是提供mint和burn接口，供DscEngine使用。后续称该ERC20 token为DSC。

    • mint: 在逻辑上负责铸造token，_mint调用负责将token从0x0发送到指定地址(user)
    function mint(address _to, uint256 _amount) external onlyOwner returns (bool) {
    if (_to == address(0)) {
        revert DecentralizedStableCoin__NotZeroAddress();
    }
    if (_amount <= 0) {
        revert DecentralizedStableCoin__MustBeMoreThanZero();
    }
    _mint(_to, _amount);
    return true;
}
    • burn: 在逻辑上负责销毁token，_burn调用本质上讲msg.sender的token还给0x0
    function mint(address _to, uint256 _amount) external onlyOwner returns (bool) {
    if (_to == address(0)) {
        revert DecentralizedStableCoin__NotZeroAddress();
    }
    if (_amount <= 0) {
        revert DecentralizedStableCoin__MustBeMoreThanZero();
    }
    _mint(_to, _amount);
    return true;
}

    3 构建DscEngine(Defi)

    代码：DscEngine

    为了实现DscEngine，要实现以下核心功能:

    • 质押token，并根据质押物的价值铸造DSC
    • 取回质押的token，并销毁相应的DSC
    • 清算liquidate，以此来维持整个Defi的正常运作

    3.1 构造函数

    需要传入:

    • 作为质押物的token地址(weth和wbtc的地址)
    • 喂价地址: 用于换算质押物token和美元的汇率
    • dsc地址(刚刚构建的ERC20的地址)
    constructor(address[] memory tokenAddresses, address[] memory priceFeedAddresses, address dscAddress) {
    if (tokenAddresses.length != priceFeedAddresses.length) {
        revert DSCEngine__TokenAddressAndPriceFeedAddressMustBeEqualLength();
    }
    for (uint256 i = 0; i < tokenAddresses.length; i++) {
        s_priceFeeds[tokenAddresses[i]] = priceFeedAddresses[i];
        s_collateralTokens.push(tokenAddresses[i]);
    }
    i_dsc = DecentralizedStableCoin(dscAddress);
}

    3.2 HealthFactor

    检测用户的health factor是整个稳定币DeFi能运作的核心。当用户的的health factor低于一定阈值时，我们就要让其他用户对其进行liquidate。

    health factor具体而言就是用户铸造的DSC和质押进去的token的比值，在这个合约中设定为1:2来满足整个Defi始终保持over-collateralized，例如质押$200价值的token则最多可以在defi中铸造$100价值的DSC。代码如下：

    function _healthFactor(address user) private view returns (uint256) {
    // total DSC minted
    // total collateral deposited
    (uint256 totalDscMinted, uint256 totalCollateralInUsd) = _getAccountInformation(user);
    if (totalDscMinted == 0) return type(uint256).max; // wtf
    uint256 collateralAdjustedForThreshold = totalCollateralInUsd * LIQUIDATION_THRESHOLD / LIQUIDATION_PRECISION;
    // 200% over-collateralized
    return (collateralAdjustedForThreshold * PRECISON) / totalDscMinted;
}

    为了计算上面代码中的totalCollateralInUsd，需要使用chainlink的预言机:Chainlink Oracle

    function getUsdValue(address token, uint256 amount) public view returns(uint256) {
    AggregatorV3Interface priceFeed = AggregatorV3Interface(s_priceFeeds[token]);
    ( , int price, , , ) = priceFeed.stablePriceCheck();
    // 从chainlink返回的价格是8位小数
    return uint256(price) * ADDITIONAL_FEED_PRECISION * amount / PRECISON;
}

    还要注意if (totalDscMinted == 0) return type(uint256).max;。即用户只选择存入质押物不去铸造DSC这当然也是可行的，要特殊判断。这是通过后续fuzz test找到的bug，会在后面再次提到。

    3.2 存入质押物并给msg.sender铸造DSC

    3.2.1 存入质押物

    逻辑比较简单，不赘述。有一行_revertIfHealthFactorIsBroken(msg.sender);即检测用户的health factor，按理来说存入质押物只会让用户的health factor更健康，可能是不必要的，删了会更省gas fee。但出于安全性考虑还是添加。

    /**
 * @param tokenCollateral 作为抵押物的token的地址
 * @param amoutCollateral 抵押物的数量
 */
function depositCollateral(address tokenCollateral, uint256 amoutCollateral)
    public
    moreThanZero(amoutCollateral)
    isAllowedToken(tokenCollateral)
    nonReentrant // 不可重入(more gas fee, safer)
{
    s_collateralDeposited[msg.sender][tokenCollateral] += amoutCollateral;
    emit CollateralDeposited(msg.sender, tokenCollateral, amoutCollateral);
    bool success = IERC20(tokenCollateral).transferFrom(msg.sender, address(this), amoutCollateral);
    if(!success) {
        revert DSCEngine__TransferFailed();
    }
    _revertIfHealthFactorIsBroken(msg.sender);
    // console.log("now balance: ", IERC20(tokenCollateral).balanceOf(address(this)));
}

    3.2.2 mint DSC

    也是注意检查health factor即可。

    /**
 * @param amountDscToMint 铸造的DSC数量
 * @notice collateralization value > minted DSC value
 */
function mintDsc(uint256 amountDscToMint)
    public 
    moreThanZero(amountDscToMint)
    nonReentrant
{
    s_dscMinted[msg.sender] += amountDscToMint;
    // if minted too much, revert(e.g. $150 DSC, $100 collateral)
    _revertIfHealthFactorIsBroken(msg.sender);
    bool minted = i_dsc.mint(msg.sender, amountDscToMint);
    if(!minted){
        revert DSCEngine__MintFailed();
    }
}

    3.3 取回质押物并销毁DSC

    3.3.1 取回质押物

    这里同样需要关注health factor。

    //取回质押物，需要health factor > 1
function redeemCollateral(address tokenCollateralAddress, uint256 amountCollateral)
    public 
    moreThanZero(amountCollateral)
    isAllowedToken(tokenCollateralAddress)
    nonReentrant
{
    _redeemCollateral(tokenCollateralAddress, amountCollateral, msg.sender, msg.sender);
    _revertIfHealthFactorIsBroken(msg.sender);
}

    function _redeemCollateral(address tokenCollateralAddress, uint256 amountCollateral,
address from, address to) private {
    s_collateralDeposited[from][tokenCollateralAddress] -= amountCollateral;
    emit CollateralRedeemed(from, to, amountCollateral, tokenCollateralAddress);
    bool success = IERC20(tokenCollateralAddress).transfer(to, amountCollateral);
    if(!success) {
        revert DSCEngine__TransferFailed();
    }
}

    3.3.2 销毁DSC

    考虑到销毁DSC只会让health factor更健康，因此check health factor这一步也可能是非必要的。出于安全考虑还是加上。

    function burnDsc(uint256 amountDscToBurn)
    public 
    moreThanZero(amountDscToBurn)
{
    _burnDsc(amountDscToBurn, msg.sender, msg.sender);
    _revertIfHealthFactorIsBroken(msg.sender); // I dont think this is necessary
}

    function _burnDsc(uint256 amountDscToBurn, address onBehalfOf, address dscFrom) private {
    s_dscMinted[onBehalfOf] -= amountDscToBurn;
    bool success = i_dsc.transferFrom(dscFrom, address(this), amountDscToBurn);
    if(!success) {
        revert DSCEngine__TransferFailed();
    }
    i_dsc.burn(amountDscToBurn); // only owner can burn
}

    3.3.3 整合逻辑

    /**
 * @param tokenCollateralAddress 质押物的地址
 * @param amountCollateral 质押物的数量
 * @param amountDscToBurn 销毁的DSC数量
 */
function redeemCollateralForDsc(
    address tokenCollateralAddress,
    uint256 amountCollateral, uint256 amountDscToBurn
) external {
    burnDsc(amountDscToBurn);
    redeemCollateral(tokenCollateralAddress, amountCollateral);
}

    3.4 清算

    先看整个代码，大致逻辑就是under-collateralized的用户会被其他用户清算，其他用户通过自己支付一定数额的DSC来销毁被清算用户的DSC，并获得被清算用户的质押物(会有10%的bonus来驱动他们清算)。

    这样做的目的是保持整个Defi是over-collateralized的(合约中存储的质押物的价值 > minted DSC的价值)。比如被清算用户质押了$200的eth来获得$100的DSC，但是过段时间他质押的eth贬值到了$180，这时根据health factor的计算，他已经under-collateralized了，其他用户可以清算。可以看到整个系统目前有$10的under-collateralized需要恢复，那么其他用户支付了$10的DSC,就可以获得对应价值$10*1.1的eth，并销毁了被清算用户$10的DSC，此时被清算用户又回到了健康状态(质押$180的eth，还有$90的DSC未被销毁)。

    // 检测某人是否under-collateralized，如果是，就以打折的形式liquidate
/**
 * 
 * @param collateral erc20 token address
 * @param user under-collateralized user(health factor < 1)
 * @param debtToCover The amount of DSC you want to burn to improve the user's health factor
 * @notice 必备条件: over-collateralized(合约中存储的质押物的价值 > minted DSC的价值)
 */
function liquidate(address collateral, address user, uint256 debtToCover) 
    external 
    moreThanZero(debtToCover)
    isAllowedToken(collateral)
    nonReentrant
{
    uint256 startingUserHealthFactor = _healthFactor(user);
    if(startingUserHealthFactor >= MIN_HEALTH_FACTOR) {
        revert DSCEngine__HealthFactorFine();
    }

    // 拿走质押物
    uint256 tokenAmountFromDebtCovered = getTokenAmountFromUsd(collateral, debtToCover);
    // 给liquidator 10% bonus
    uint256 bonusCollateral = (tokenAmountFromDebtCovered * LIQUIDATION_BONUS) / LIQUIDATION_PRECISION;
    uint256 totalCollateralToRedeem = tokenAmountFromDebtCovered + bonusCollateral;
    _redeemCollateral(collateral, totalCollateralToRedeem, user, msg.sender);

    // 销毁DSC,msg.sender支付,user的DSC被销毁
    _burnDsc(debtToCover, user, msg.sender);

    uint256 endingUserHealthFactor = _healthFactor(user);
    if(endingUserHealthFactor <= startingUserHealthFactor) {
        revert DSCEngine__HealthFactorNotImproved();
    }
    _revertIfHealthFactorIsBroken(msg.sender);
}

    这里其实是一个比较复杂的模型，我们不能保证这个模型能够始终保持over-collateralized，作为一个Defi需要通过安全审计。这里简单做了Invariant测试，保证模型始终有合约中存储的质押物的价值 > minted DSC的价值。

    5 Oracle Protection

    用了chainlink的预言机，这其实不是完成去中心化，如果chainlink的预言机失效了，这个Defi也会失效。我们需要在喂价一定时间不更新时抛出异常。

    library OracleLib {
    error OracleLib__StablePrice();
    uint256 private constant TIMEOUT = 1 hours;
    function stablePriceCheck(AggregatorV3Interface priceFeed) 
        public 
        view 
        returns (uint80, int256,uint256, uint256, uint80) 
    {
        (uint80 roundId, int256 price, uint256 startedAt, uint256 updatedAt, uint80 answeredInRound) = priceFeed.latestRoundData();
        uint256 secondsSinceLastUpdate = block.timestamp - updatedAt;
        if(secondsSinceLastUpdate > TIMEOUT){
            // 如果超过1小时没有更新，冻结DscEngine
            revert OracleLib__StablePrice();
        }
        return (roundId, price, startedAt, updatedAt, answeredInRound);
    }
}
